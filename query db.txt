create table client
(
cedula int primary key,
name varchar(20), 
lastname varchar(20),  
age int, 
gender varchar(60),
height int,
weight int,
phone int,  
start_date varchar(10), 
end_date varchar(10),
bmi numeric(4, 2)
);


create table expense
(
id int serial primary key,
value int, 
kind_value varchar(20),  
date varchar(10)
);

create table revenue
(
id int serial primary key,
value int, 
kind_value varchar(20),  
date varchar(10)
);
